72:     from_time = from_time.to_time if from_time.respond_to?(:to_time)
73:     to_time = to_time.to_time if to_time.respond_to?(:to_time)
74:     distance_in_minutes = (((to_time - from_time).abs)/60).round
75:     I18n.with_options :locale => options[:locale], :scope => 'datetime.distance_in_words' do |locale|
76:     case distance_in_minutes
77:     when 0..1           then (distance_in_minutes==0) ? (locale.t :less_than_a_minute, :count => 5) : (locale.t :one_minute_ago, :count => distance_in_minutes)
78:     when 2..59          then locale.t :x_minutes_ago, :count => distance_in_minutes 
79:     when 60..90         then locale.t :one_hour_ago, :count => distance_in_minutes
80:     when 90..1440       then locale.t :x_hours_ago, :count => (distance_in_minutes.to_f / 60.0).round
81:     when 1440..2160     then locale.t :one_day_ago, :count => distance_in_minutes # 1 day to 1.5 days
82:     when 2160..2880     then locale.t :x_days_ago, :count => (distance_in_minutes.to_f / 1440.0).round # 1.5 days to 2 days
83:     else I18n.l from_time, :format => :long1
84:     end
85:   end
